void generarRandom(int v[], int size)
{
     int i, j, num, dupl;
 
     //genero mis numeros al azar
     for (i = 0; i < size; i++){
         num = 1 + rand() % size;
         dupl = 0;
         
         for (j = 0; j <= i; j++){
             if (num == v[j]){
                dupl = 1;
                break;
             }
         }
         if (dupl == 1)
            i--;
         else
            v[i] = num;
     }
}
